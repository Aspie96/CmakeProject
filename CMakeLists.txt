CMAKE_MINIMUM_REQUIRED (VERSION 3.16 FATAL_ERROR)

PROJECT ("parallel00_v1d" LANGUAGES CXX CUDA)

INCLUDE(FindCUDA)

CUDA_SELECT_NVCC_ARCH_FLAGS(CUDA_ARCH_FLAGS)

ADD_EXECUTABLE ("parallel00_v1d" "parallel00_v1d.cu" )

if (CMAKE_CXX_COMPILER_ID STREQUAL "GNU")
	set(CMAKE_CXX_FLAGS_RELEASE "-O2")
	set(CMAKE_C_FLAGS_RELEASE "-O2")
endif()

MESSAGE($<$<COMPILE_LANGUAGE:CUDA>:${CUDA_ARCH_FLAGS}>)

TARGET_COMPILE_OPTIONS("parallel00_v1d" PRIVATE $<$<COMPILE_LANGUAGE:CUDA>:${CUDA_ARCH_FLAGS}>)
TARGET_INCLUDE_DIRECTORIES("parallel00_v1d" PRIVATE ${CMAKE_CUDA_TOOLKIT_INCLUDE_DIRECTORIES})
