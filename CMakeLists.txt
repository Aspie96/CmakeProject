CMAKE_MINIMUM_REQUIRED (VERSION 3.17 FATAL_ERROR)

PROJECT ("parallel" LANGUAGES CXX CUDA)
PROJECT ("serial" LANGUAGES C CXX)

INCLUDE(FindCUDA)

CUDA_SELECT_NVCC_ARCH_FLAGS(CUDA_ARCH_FLAGS)
ADD_EXECUTABLE ("parallel" "parallel.cu" )
ADD_EXECUTABLE ("serial" "serial.c" )
TARGET_COMPILE_OPTIONS("parallel" PRIVATE $<$<COMPILE_LANGUAGE:CUDA>:${CUDA_ARCH_FLAGS}>)
TARGET_INCLUDE_DIRECTORIES("parallel" PRIVATE ${CMAKE_CUDA_TOOLKIT_INCLUDE_DIRECTORIES})
