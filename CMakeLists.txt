CMAKE_MINIMUM_REQUIRED (VERSION 3.17 FATAL_ERROR)

PROJECT ("MyProject" LANGUAGES CXX CUDA)

INCLUDE(FindCUDA)

CUDA_SELECT_NVCC_ARCH_FLAGS(CUDA_ARCH_FLAGS)
add_executable ("MyProject" "CMakeProject2.cu" )
TARGET_COMPILE_OPTIONS("MyProject" PRIVATE $<$<COMPILE_LANGUAGE:CUDA>:${CUDA_ARCH_FLAGS}>)
TARGET_INCLUDE_DIRECTORIES("MyProject" PRIVATE ${CMAKE_CUDA_TOOLKIT_INCLUDE_DIRECTORIES})
